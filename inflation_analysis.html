  invisible(lapply(packages, library, character.only = TRUE))
})

fredr_set_key(Sys.getenv("FRED_API_KEY"))

start_date <- as.Date("2021-01-01")

fetch_fred_data <- function(series_id, start_date, end_date = Sys.Date()) {
  if (as.Date(end_date) < as.Date(start_date)) {
    stop("end_date must be after start_date")
  }
  fredr(
    series_id = series_id,
    observation_start = start_date,
    observation_end = as.Date(end_date)
  )
}

4 Demand Pressure

Code 
# Fetch raw CPI and earnings data
df_cpi_raw <- fetch_fred_data("CPIAUCSL", start_date) %>%
  select(date, cpi = value)

df_earn <- fetch_fred_data("CES0500000003", start_date) %>%
  select(date, earnings = value)

# Real wages
df_rw <- merge(df_cpi_raw, df_earn, by = "date") %>%
  mutate(
    real_wage = earnings / cpi,
    growth = 100 * (real_wage - lag(real_wage, 12)) / lag(real_wage, 12)
  ) %>%
  filter(!is.na(growth)) %>%
  filter(month(date) %in% c(1, 4, 7, 10)) %>%
  mutate(date = floor_date(date, "quarter")) %>%
  select(date, value = growth) %>%
  mutate(series = "Real Wage Growth")

# Productivity - values are ALREADY YoY % change, do not recompute
df_prod <- fetch_fred_data("PRS85006091", start_date) %>%
  select(date, value) %>%
  filter(!is.na(value)) %>%
  mutate(series = "Labor Productivity")

# Combine datasets
dat_dp <- bind_rows(df_rw, df_prod) %>%
  mutate(date = floor_date(date, "quarter")) %>%
  filter(date >= as.Date("2022-01-01"))

# Plot
ggplot(dat_dp, aes(date, value, color = series)) +
  
  geom_hline(yintercept = 0, linetype = "dashed", color = "black") +
  
  geom_line(linewidth = 1.4) +
  
  scale_color_manual(values = c(
    "Real Wage Growth" = "purple",
    "Labor Productivity" = "steelblue"
  )) +
  
  labs(
    title = "Real Wage Growth & Labor Productivity",
    subtitle = "Year-over-year percent change",
    x = "",
    y = "Year-over-Year % Change",
    color = "",
    caption = "Source: BLS via FRED®"
  ) +
  
  scale_x_date(
    date_breaks = "6 months",
    date_labels = "%b %Y",
    expand = expansion(mult = c(0.02, 0.05))
  ) +
  
  coord_cartesian(xlim = c(as.Date("2023-01-01"), NA)) +
  
  theme_minimal() +
  theme(legend.position = "top")

Real wage growth has recovered sharply from deeply negative levels and is converging toward productivity growth, suggesting building demand-side inflationary pressure.

6 Outlook

While easing energy prices may continue to reduce headline inflation, renewed volatility in oil markets introduces upside risk. Converging wage and productivity growth, alongside elevated core inflation, suggest inflation is likely to decline gradually rather than rapidly.

Looking ahead, key indicators to monitor include:

  • Energy prices, particularly oil, given their sensitivity to geopolitical developments
  • Core inflation measures, which reflect underlying price pressures
  • Wage growth relative to productivity, as a signal of demand-side inflation
  • Financial market indicators, such as the yield curve, which provide insight into expectations for growth and inflation

Overall, while inflation is expected to continue moderating, risks remain tilted toward a slower and uneven path back to target.
