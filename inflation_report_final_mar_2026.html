start_date <- as.Date("2021-01-01")

fetch_fred_data <- function(series_id, start_date, end_date = Sys.Date()) {
  if (as.Date(end_date) < as.Date(start_date)) {
    stop("end_date must be after start_date")
  }
  fredr(
    series_id = series_id,
    observation_start = start_date,
    observation_end = as.Date(end_date)
  )
}

df_cpi <- fetch_fred_data("CPIAUCSL", start_date) %>%
  select(date, value) %>%

4 Demand Pressure

# Real wages
df_cpi_raw <- fetch_fred_data("CPIAUCSL", start_date) %>% select(date, cpi=value)
df_earn <- fetch_fred_data("CES0500000003", start_date) %>% select(date, earnings=value)

df_rw <- merge(df_cpi_raw, df_earn, by="date") %>%
  mutate(real_wage = earnings / cpi,
         growth = 100 * (real_wage - lag(real_wage,12)) / lag(real_wage,12)) %>%
  filter(!is.na(growth)) %>%
  mutate(q = floor_date(date,"quarter")) %>%
  group_by(q) %>%
  summarise(value = mean(growth)) %>%
  rename(date=q) %>%
  mutate(series="Real Wage Growth")

# Productivity
df_prod <- fetch_fred_data("PRS85006091", start_date) %>%
  select(date,value) %>%
  mutate(value = 100 * (value - lag(value,4)) / lag(value,4)) %>%
  filter(!is.na(value)) %>%
  mutate(series="Labor Productivity")

#Combine datasets
dat_dp <- bind_rows(df_rw, df_prod) %>%
  filter(date >= as.Date("2022-01-01"))

# Alignment
dat_diff <- dat_dp %>%
  pivot_wider(names_from = series, values_from = value) %>%
  arrange(date) %>%
  mutate(diff = `Real Wage Growth` - `Labor Productivity`) %>%
  filter(!is.na(diff))

# Plot

ggplot(dat_diff, aes(date, diff)) +
  
  geom_hline(yintercept = 0, linetype = "dashed", color = "black") +
  
  geom_line(color = "purple", linewidth = 1.4) +
  
  labs(
    title = "Wage Growth Relative to Productivity",

6 Outlook

While easing energy prices may continue to reduce headline inflation, renewed volatility in oil markets introduces upside risk. Persistent wage growth and elevated core inflation suggest that inflation is likely to decline gradually rather than rapidly. Looking ahead, key indicators to monitor include:

  • Energy prices, particularly oil, given their sensitivity to geopolitical developments
  • Core inflation measures, which reflect underlying price pressures
  • Wage growth relative to productivity, as a signal of demand-side inflation
  • Financial market indicators, such as the yield curve, which provide insight into expectations for growth and inflation

Overall, while inflation is expected to continue moderating, risks remain tilted toward a slower and uneven path back to target.
